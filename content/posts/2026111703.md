**第二步：开始调用GPT-4o**
java
import okhttp3.*;
import com.google.gson.JsonObject;
import com.google.gson.JsonParser;

import java.io.IOException;

public class QianjuAIExample {

    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    // 这里的key，去[千聚ai聚合平台](https://www.qianjuai.com/)（www.qianjuai.com）后台复制过来
    private static final String API_KEY = "sk-你的千聚APIKey";

    public static void main(String[] args) throws IOException {
        OkHttpClient client = new OkHttpClient();

        // 构建请求体
        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", "用中文写一首关于人工智能发展的五言绝句。");

        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "gpt-4o");
        requestBody.add("messages", new com.google.gson.JsonArray());
        requestBody.getAsJsonArray("messages").add(message);
        requestBody.addProperty("max_tokens", 500);
        requestBody.addProperty("temperature", 0.7);

        Request request = new Request.Builder()
                .url(API_URL)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(
                        requestBody.toString(),
                        MediaType.parse("application/json; charset=utf-8")))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                System.out.println("请求失败， HTTP状态码: " + response.code());
                return;
            }
            if (response.body() == null) {
                System.out.println("响应体为空");
                return;
            }
            String responseBody = response.body().string();
            JsonObject jsonResponse = JsonParser.parseString(responseBody).getAsJsonObject();
            String content = jsonResponse
                    .getAsJsonArray("choices")
                    .get(0)
                    .getAsJsonObject()
                    .get("message")
                    .getAsJsonObject()
                    .get("content")
                    .getAsString();
            System.out.println("GPT-4o回复: ");
            System.out.println(content);
        }
    }
}

你看，就这几行代码，一个能正常调用GPT-4o的Java程序就完工了。不需要额外配置，不需要代理，也不需要改任何网络设置。当初我为了搞懂OpenAI代理配置，可是白白折腾了好几小时。
