**完整示例（Python + Flask接收钉钉消息）：**

python
from flask import Flask, request
import openai

app = Flask(__name__)

# 千聚配置
openai.api_base = "https://www.qianjuai.com/v1"
openai.api_key = "sk-千聚申请的Key"

@app.route('/dingtalk', methods=['POST'])

    data = request.json
    user_msg = data['text']['content']
    # 调用模型

        model="gpt-4o",
        messages=[{"role": "user", "content": user_msg}]
    )
    reply = response.choices[0].message.content
    # 返回给钉钉
    return {"msgtype": "text", "text": {"content": reply}}
