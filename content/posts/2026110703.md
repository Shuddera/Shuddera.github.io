java
import io.github.openai.OpenAiApi;
import io.github.openai.model.ChatCompletionRequest;
import io.github.openai.model.ChatMessage;
import io.github.openai.model.ChatMessageRole;
import okhttp3.OkHttpClient;
import retrofit2.Retrofit;
import retrofit2.adapter.rxjava2.RxJava2CallAdapterFactory;
import retrofit2.converter.gson.GsonConverterFactory;

import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.TimeUnit;

public class QwenTurboJavaExample {
    public static void main(String[] args) {
        // 替换成你在[[千聚ai官网](https://www.qianjuai.com/)](https://www.qianjuai.com/)申请的API Key
        String apiKey = "sk-你的千聚API密钥";

        // ✅ 关键一步：将base_url改为千聚的接口地址
        String baseUrl = "https://www.qianjuai.com/v1";

        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .writeTimeout(60, TimeUnit.SECONDS)
                .build();

        // 创建Retrofit实例
        Retrofit retrofit = new Retrofit.Builder()
                .baseUrl(baseUrl)
                .client(client)
                .addConverterFactory(GsonConverterFactory.create())
                .addCallAdapterFactory(RxJava2CallAdapterFactory.create())
                .build();

        OpenAiApi openAiApi = retrofit.create(OpenAiApi.class);

        // 构建请求（调用 Qwen-Turbo 模型）
        List<ChatMessage> messages = new ArrayList<>();
        messages.add(new ChatMessage(ChatMessageRole.SYSTEM.value(), "你是一个专业的Java开发助手。"));
        messages.add(new ChatMessage(ChatMessageRole.USER.value(), "请用Java写一个简单的for循环，打印1到10。"));

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("qwen-turbo") // 关键：模型名填 qwen-turbo
                .messages(messages)
                .temperature(0.7)
                .build();

        try {
            // 发起同步请求
            ChatCompletionResult result = openAiApi.createChatCompletion("Bearer " + apiKey, request).execute().body();
            
            if (result != null && result.getChoices() != null && !result.getChoices().isEmpty()) {
                String content = result.getChoices().get(0).getMessage().getContent();
                System.out.println("Qwen-Turbo 响应内容:\n" + content);
            } else {
                System.out.println("请求失败，请检查API Key或网络连接。");
            }
        } catch (Exception e) {
            System.err.println("发生异常:" + e.getMessage());
        }
    }
}


**解释：**
*   **`baseUrl`**：这是唯一需要改动的地方，指向千聚ai的网关 `https://www.qianjuai.com/v1`。
*   **`model`**：直接指定为 `"qwen-turbo"`，千聚会自动处理路由。无需关心千聚后台具体调用了哪个渠道。
*   **稳定性**：代码中设置了60秒的超时，足以应对流式或非流式响应。

其他任何基于OpenAI SDK的代码，**只需要改这一行 `base_url`，就能接上Qwen-Turbo**。你的LangChain、LlamaIndex等框架代码，也适用这个规则。
