### Java代码示例（用OkHttp）

java
import okhttp3.*;
import org.json.JSONObject;
import java.io.IOException;

public class ChatGLMExample {
    public static void main(String[] args) {
        // [千聚api聚合站](https://www.qianjuai.com/)的API地址，完全兼容OpenAI格式
        String baseUrl = "https://www.qianjuai.com/v1";
        // 你的千聚API Key
        String apiKey = "sk-你的千聚API Key";

        OkHttpClient client = new OkHttpClient().newBuilder().build();
        MediaType mediaType = MediaType.parse("application/json");

        // 构造请求体，模型名直接用智谱清言的glm-4（千聚支持）
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "glm-4"); // 这里可换成 glm-3-turbo 等
        requestBody.put("messages", new JSONObject[]{
            new JSONObject().put("role", "system").put("content", "你是一个专业助教，请简洁回答。"),
            new JSONObject().put("role", "user").put("content", "Java中如何实现单例模式？")
        });

        Request request = new Request.Builder()
            .url(baseUrl + "/chat/completions")
            .method("POST", RequestBody.create(mediaType, requestBody.toString()))
            .addHeader("Authorization", "Bearer " + apiKey)
            .addHeader("Content-Type", "application/json")
            .build();

        try {
            Response response = client.newCall(request).execute();
            String responseBody = response.body().string();
            System.out.println("智谱清言回复：");
            System.out.println(new JSONObject(responseBody)
                .getJSONArray("choices")
                .getJSONObject(0)
                .getJSONObject("message")
                .getString("content"));
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
