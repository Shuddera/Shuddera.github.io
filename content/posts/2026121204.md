    api_key="sk-your-qianju-api-key-here",

# 2. 调用最新 GPT-5.1 模型

    model="gpt-5.1",  # 直接写模型名即可
    messages=[
        {"role": "system", "content": "你是一个聪明的AI助手。"},
        {"role": "user", "content": "用中文写一段关于AI未来的简短论述。"}
    ],
    max_tokens=1024,
    stream=True  # 支持流式输出

# 3. 流式输出结果

    if chunk.choices[0].delta.content:
        print(chunk.choices[0].delta.content, end="")
