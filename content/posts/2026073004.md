### 第二步：编写接入代码

java
import okhttp3.*;
import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONObject;
import java.io.IOException;

public class DeepSeekClient {
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    private static final String API_KEY = "你的千聚API_Key";

    public static void main(String[] args) throws IOException {
        OkHttpClient client = new OkHttpClient();

        JSONObject message = new JSONObject();
        message.put("role", "user");
        message.put("content", "用Java写一个斐波那契数列的递归方法");

        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "deepseek-chat"); // 对应DeepSeek-V3
        requestBody.put("messages", new JSONObject[]{message});

        Request request = new Request.Builder()
                .url(API_URL)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        requestBody.toJSONString()
                ))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                String json = response.body().string();
                JSONObject jsonObject = JSON.parseObject(json);
                String result = jsonObject.getJSONArray("choices")
                        .getJSONObject(0)
                        .getJSONObject("message")
                        .getString("content");
                System.out.println("DeepSeek回复：" + result);
            } else {
                System.out.println("请求失败：" + response.code() + " " + response.message());
            }
        }
    }
}
