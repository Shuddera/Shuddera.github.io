**核心代码示例：**

import openai

# 将 base_url 改为千聚的接口地址
client = openai.OpenAI(
    api_key="你的千聚API Key",  # 在千聚控制台创建

# 调用 Gemini 2.0 Flash 模型
response = client.chat.completions.create(
    model="gemini-2.0-flash",  # 千聚支持的模型名称，见文档
    messages=[
        {"role": "system", "content": "你是一个Python开发助手。"},
        {"role": "user", "content": "用Python写一个生成斐波那契数列的函数。"}
    ],
    stream=True  # 支持流式输出
)

for chunk in response:
    print(chunk.choices[0].delta.content or "", end="")
