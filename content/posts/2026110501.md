public class 豆包统一接入示例 {

    public static void main(String[] args) {
        // 原版：使用 OpenAI 官方的 base_url

        // 构建客户端

        );

        // 构建消息
        Message systemMessage = new Message("system", "你是一个优秀的 Java 编程助手。");
        Message userMessage = new Message("user", "请用 Java 写一个单例模式示例。");

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-4o") // 这里想调用哪个模型就填哪个模型名
                .messages(Arrays.asList(systemMessage, userMessage))
                .maxTokens(500)
                .build();

        // 调用
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }

    // 一个自定义的拦截器，用于注入千聚的认证头（示例）
    static class ApiAuthInterceptor implements Interceptor {
        private final String apiKey;

        public ApiAuthInterceptor(String apiKey) {
            this.apiKey = apiKey;
        }

        @Override
        public Response intercept(Chain chain) throws IOException {
            Request.Builder builder = chain.request().newBuilder();
            builder.header("Authorization", "Bearer " + apiKey);
            return chain.proceed(builder.build());
        }
    }
}
