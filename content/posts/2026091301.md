import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;

public class Qwen3Demo {
    public static void main(String[] args) {
        // 替换成你的千聚API Key

        // 注意：SDK默认base_url是https://api.openai.com/v1
        // 需要手动指定baseUrl，具体方式取决于SDK版本
        // 这里用OkHttpClient方式演示（实际项目中建议用配置文件）

        // 构造消息
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("qwen3-72b")  // 选择Qwen3模型
                .messages(List.of(
                        new ChatMessage("system", "你是一个专业的Java技术顾问。"),
                        new ChatMessage("user", "请用Java写一个简单的REST接口，返回当前时间。")
                ))
                .maxTokens(500)
                .temperature(0.7)
                .build();

        // 发送请求
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
